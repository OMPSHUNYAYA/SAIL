(() => {
  SAIL_REQ.reset();

  const initial = SAIL_REQ.current();
  const security = SAIL_REQ.loadExample("security");
  const incident = SAIL_REQ.loadExample("incident");
  const pack = SAIL_REQ.exportPack();

  const structural =
    pack.structural_payload.requirements;

  const operational =
    pack.operational_metadata.records;

  const records =
    operational.map(item => item.record_id);

  const recordNumbers =
    records
      .map(record =>
        Number(record.replace("REQREC-",""))
      )
      .sort((first,second) => first - second);

  const requirementPreserved =
    requirement =>
      structural.some(
        item =>
          item.requirement_id ===
            requirement.requirement_id &&
          item.title === requirement.title
      ) &&
      operational.some(
        item =>
          item.requirement_id ===
            requirement.requirement_id &&
          item.record_id === requirement.record_id
      );

  return {
    requirement_count:
      Number(
        document.getElementById("reqCount").textContent
      ),

    titles:
      structural
        .map(item => item.title)
        .sort(),

    unique_record_ids:
      new Set(records).size === records.length,

    monotonic_record_ids:
      recordNumbers.every(
        (value,index,array) =>
          index === 0 ||
          value === array[index - 1] + 1
      ),

    initial_requirement_preserved:
      requirementPreserved(initial),

    security_requirement_preserved:
      requirementPreserved(security),

    incident_requirement_preserved:
      requirementPreserved(incident)
  };
})()
